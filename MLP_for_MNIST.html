creating sequential model¶

In [81]:
model = Sequential([
	
	# reshape 28 row * 28 column data to 28*28 rows
	Flatten(input_shape=(28, 28)),
	
	# dense layer 1
	Dense(256, activation='sigmoid'),
	
	# dense layer 2
	Dense(128, activation='sigmoid'),
	
	# output layer
	Dense(10, activation='sigmoid'),
])
